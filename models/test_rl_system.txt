def test_learning_memory():
    """Test the learning memory system."""
    print("=== Testing Learning Memory System ===")
    
    # Create sample experiences
    experiences = create_sample_experiences()
    
    # Create learning memory
    memory = LearningMemory(experiences=[], model_performance={})
    
    # Add experiences
    for exp in experiences:
        memory.add_experience(exp)
        print(f"Added experience: {exp.error_code} - Success: {exp.success} - Reward: {exp.reward:.1f}")
    
    # Test memory functions
    print(f"\nTotal experiences: {len(memory.experiences)}")
    print(f"Overall success rate: {memory.get_success_rate():.2%}")
    
    # Test error code specific queries
    for error_code in ["D401", "N802", "F821", "D103"]:
        exp_count = len(memory.get_experiences_by_error_code(error_code))
        if exp_count > 0:
            success_rate = memory.get_success_rate(error_code)
            print(f"{error_code}: {exp_count} experiences, {success_rate:.2%} success rate")
    
    return memory

def test_rl_manager():
    """Test the RL manager system."""
    print("\n=== Testing RL Manager System ===")
    
    # Create RL manager
    rl_manager = RLManager()
    
    # Add sample experiences
    experiences = create_sample_experiences()
    for exp in experiences:
        rl_manager.record_experience(exp)
    
    # Test learning prompts
    print("\n--- Learning Prompts ---")
    for error_code in ["D401", "N802", "F821", "D103"]:
        prompt = rl_manager.get_learning_prompt(error_code, "Test error message")
        if prompt:
            print(f"{error_code}: {prompt[:100]}...")
        else:
            print(f"{error_code}: No learning feedback available yet")
    
    # Test performance summary
    print("\n--- Performance Summary ---")
    summary = rl_manager.get_performance_summary()
    print(f"Total experiences: {summary['total_experiences']}")
    print(f"Overall success rate: {summary['overall_success_rate']:.2%}")
    print(f"Error code performance:")
    for error_code, success_rate in summary['error_code_performance'].items():
        print(f"  {error_code}: {success_rate:.2%}")
    
    return rl_manager

def test_quality_verification():
    """Test the quality verification system."""
    print("\n=== Testing Quality Verification System ===")
    
    # Test cases
    test_cases = [
        {
            "name": "Valid fix",
            "original": 'def test():\n    """Test function."""\n    pass',
            "fixed": 'def test():\n    """Test function."""\n    pass',
            "response": 'def test():\n    """Test function."""\n    pass'
        },
        {
            "name": "Fix with explanation",
            "original": 'def test():\n    """Test function."""\n    pass',
            "fixed": 'def test():\n    """Test function."""\n    pass',
            "response": 'def test():\n    """Test function."""\n    pass\n\n# Fixed the docstring'
        },
        {
            "name": "Syntax error",
            "original": 'def test():\n    pass',
            "fixed": 'def test():\n    pass\n    return',  # Invalid syntax
            "response": 'def test():\n    pass\n    return'
        }
    ]
    
    for test_case in test_cases:
        success, reward, failure_reason = verify_fix_quality(
            test_case["original"],
            test_case["fixed"],
            test_case["response"]
        )
        print(f"{test_case['name']}: Success={success}, Reward={reward:.1f}, Failure={failure_reason}")

def main():
    """Run all tests."""
    print("🧠 LLM Ruff Fixer - Reinforcement Learning System Test")
